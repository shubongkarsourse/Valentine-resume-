// Music toggle
function toggleMusic(id){
  let m=document.getElementById(id);
  let btn=document.getElementById(id+'Btn');
  if(m.paused){
    m.play().catch(()=>{});
    btn.innerHTML='ðŸŽµ<span>Music ON</span>';
  } else {
    m.pause();
    btn.innerHTML='ðŸŽµ<span>Music OFF</span>';
  }
}

// Smooth Page Transition
function goToPage(n){
  document.querySelectorAll('.page').forEach(p=>{
    p.style.opacity=0;
    p.style.transform='scale(0.95)';
    setTimeout(()=>p.classList.remove('active'),300);
  });
  setTimeout(()=>{
    const pg=document.getElementById('page'+n);
    pg.classList.add('active');
    pg.style.opacity=1;
    pg.style.transform='scale(1)';
  },300);
  // Enable music after any page tap
  enableMusic();
}
